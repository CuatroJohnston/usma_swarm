74  void compute_dots()
75  {
76  // Source: http://math.stackexchange.com/questions/1472049/check-if-a-point-is-inside-a-rectangular-shaped-area-3d
77  u = points[0] - points[1];
78  v = points[0] - points[3];
79  w = points[0] - points[4];
80 
81  u_dot_P0 = u.dot(points[0]);
82  u_dot_P1 = u.dot(points[1]);
83  v_dot_P0 = v.dot(points[0]);
84  v_dot_P3 = v.dot(points[3]);
85  w_dot_P0 = w.dot(points[0]);
86  w_dot_P4 = w.dot(points[4]);
87  }
88 
89  bool contains(Eigen::Vector3d p)
90  {
91  double u_dot_p = u.dot(p);
92  double v_dot_p = v.dot(p);
93  double w_dot_p = w.dot(p);
94 
95  if ( (u_dot_P0 > u_dot_p) && (u_dot_p > u_dot_P1) &&
96  (v_dot_P0 > v_dot_p) && (v_dot_p > v_dot_P3) &&
97  (w_dot_P0 > w_dot_p) && (w_dot_p > w_dot_P4)) {
98  return true;
99  }
100  return false;
101  }
102  };
